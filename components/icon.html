使用图标 ​

  • 如果你想像用例一样直接使用，你需要全局注册组件，才能够直接在项目里使用。
vue
<script setup lang="ts">
import { CashOutline } from '@vicons/ionicons5';
</script>
<template>
  <LIcon color="red" size="40">
    <CashOutline />
  </LIcon>
</template>
